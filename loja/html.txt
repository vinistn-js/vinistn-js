    <script>
        const menuHamburguer = document.querySelector('.menu-hamburguer');
        const navLinks = document.querySelector('.nav-links.categories'); // Ajusta para a lista de categorias
        const themeIcon = document.getElementById('theme-icon');

        menuHamburguer.addEventListener('click', () => {
            navLinks.classList.toggle('active');
        });

        // Função para alternar o tema
        function toggleTheme() {
            document.body.classList.toggle('dark-mode');
            // Altera o ícone baseado no tema atual
            if (document.body.classList.contains('dark-mode')) {
                themeIcon.classList.remove('fa-sun');
                themeIcon.classList.add('fa-moon');
            } else {
                themeIcon.classList.remove('fa-moon');
                themeIcon.classList.add('fa-sun');
            }
        }

        // Adiciona o evento de clique ao ícone de tema
        themeIcon.addEventListener('click', toggleTheme);

        // Define o tema inicial com base na configuração do dispositivo
        if (window.matchMedia('(prefers-color-scheme: dark)').matches) {
            document.body.classList.add('dark-mode');
            themeIcon.classList.remove('fa-sun');
            themeIcon.classList.add('fa-moon');
        }
    </script>
